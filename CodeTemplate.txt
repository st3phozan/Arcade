if (Player runs near a Skeleton) {
    The Skeleton awakens and focuses on the character
} else {
   Skeleton continues to walk around and stays passive
}

if (Player stays near Skeleton for some time) {
    Skeleton begins to fire arrows
} else {
    Skeleton becomes passive again after Player leaves
}

if (An arrow hits the Player) {
    Player will lose health
} else {
    Player keeps health and will continue to pursue Skeleton
}
